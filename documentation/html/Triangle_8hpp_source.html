37 bool hit(const Ray &r, double t_min, double t_max, Structs::hitRecord &rec) const override
38 {
39 Vector3D edge1 = v1 - v0;
40 Vector3D edge2 = v2 - v0;
41 Vector3D h = cross(r.getDirection(), edge2);
42 double a = dot(edge1, h);
43
44 if (a > -1e-8 && a < 1e-8)
45 return false;
46
47 double f = 1.0 / a;
48 Vector3D s = r.getOrigin() - v0;
49 double u = f * dot(s, h);
50
51 if (u < 0.0 || u > 1.0)
52 return false;
53
54 Vector3D q = cross(s, edge1);
55 double v = f * dot(r.getDirection(), q);
56
57 if (v < 0.0 || u + v > 1.0)
58 return false;
59
60 double t = f * dot(edge2, q);
61
62 if (t < t_min || t > t_max)
63 return false;
64
65 rec.t = t;
66 rec.point = r.pointAtParameter(t);
67 rec.normal = normal;
68 rec.material = material;
69 rec.color = color;
70
71 return true;
72 }

74 private:
75 Point3D calculateCenter(const Point3D &v0, const Point3D &v1, const Point3D &v2) const
76 {
77 return Point3D(
78 (v0.x + v1.x + v2.x) / 3.0,
79 (v0.y + v1.y + v2.y) / 3.0,
80 (v0.z + v1.z + v2.z) / 3.0
81 );
82 }

86 Vector3D normal; // Normale au plan du triangle
