56 Vector3D getColor(const Ray &ray, int depth) {
57 Structs::hitRecord rec;
58 if (checkForHit(ray, 0.001, MAXFLOAT, rec)) {
59 Ray scattered;
60 Vector3D attenuation;
61 if (depth < 50 && rec.material->scatter(ray, rec, attenuation, scattered)) {
62 return attenuation * getColor(scattered, depth + 1);
63 } else {
64 return Vector3D(0, 0, 0);
65 }
66 } else {
67 Vector3D unit = ray.getDirection() / ray.getDirection().length();
68 double t = 0.5 * (unit.y + 1);
69 return (1 - t) * Vector3D(1,1,1);
70 }
